# OPERATOR PRECEDENCE
17  NOT, UNARY PLUS, UNARY DASH

16  EXP (right-assoc)

15  STAR, FSLASH, PERCENT (left-assoc)

14  PLUS, DASH (left-assoc)

12  LT, LE, GT, GE

11  EQUALS, NEQUALS

7   AND

6   OR

3   ASSIGN (right-assoc)


# GRAMMAR

expr            ::=     NOT expr |
                        (PLUS|DASH) expr |
                        primary-expr |
                        expr EXP expr |
                        expr (STAR|FSLASH|PERCENT) expr |
                        expr (PLUS|DASH) expr |
                        expr (LT|LE|GT|GE) expr |
                        expr EQUALS expr
                        expr AND expr |
                        expr OR expr |
                        IDENTIFIER EQ expr

primary-expr    ::=     LPAREN expr RPAREN | INTEGER | DOUBLE | (KEYWORD:true|KEYWORD:false) | STRING | IDENTIFIER
